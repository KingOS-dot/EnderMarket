// ── CURSOR ───────────────────────────────────
const dot = document.getElementById('cur');
const ring = document.getElementById('cur-ring');
let mx = 0, my = 0, rx = 0, ry = 0, alive = false;

document.addEventListener('mousemove', e => {
  mx = e.clientX; my = e.clientY;
  dot.style.transform = `translate3d(${mx - 5}px,${my - 5}px,0)`;
  if (!alive) { alive = true; dot.style.opacity = '1'; ring.style.opacity = '1'; }
});

(function loop() {
  rx += (mx - rx) * 0.14;
  ry += (my - ry) * 0.14;
  ring.style.transform = `translate3d(${rx - 20}px,${ry - 20}px,0)`;
  requestAnimationFrame(loop);
})();

function bindCursor() {
  document.querySelectorAll('a,button,.ftab,.pcard,.ccard,.hc').forEach(el => {
    el.addEventListener('mouseenter', () => ring.classList.add('big'));
    el.addEventListener('mouseleave', () => ring.classList.remove('big'));
  });
}
bindCursor();
